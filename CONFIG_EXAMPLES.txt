# ============================================
# DOCKER DEPLOYMENT
# ============================================
# Basic deployment:
#   docker-compose up -d
#
# With custom session:
#   Edit docker-compose.yml:
#   command: ["--host", "0.0.0.0", "--port", "7860", "--session", "my-event"]
#
# View logs:
#   docker-compose logs -f
#
# Stop:
#   docker-compose down

# ============================================
# NETWORK SHARING
# ============================================
# To share on local network:
#   python app.py --host 0.0.0.0 --session my-event
#
# Access from other devices:
#   http://YOUR_IP:7860
#
# Find your IP:
#   Linux/Mac: ip addr show
#   Windows: ipconfig

# ============================================
# PERFORMANCE TIPS
# ============================================
# 1. Use smaller models for faster processing
# 2. Disable interim results if not needed
# 3. Use internal translation for fastest results
# 4. Close unused browser tabs to free resources
# 5. For Docker, allocate sufficient memory (2GB+)

# ============================================
# TROUBLESHOOTING
# ============================================
# No audio input:
#   - Check microphone permissions
#   - Verify correct device selected
#   - On Linux: sudo usermod -a -G audio $USER
#
# Model not loading:
#   - Ensure model is fully extracted
#   - Check models/ directory permissions
#   - Verify model path is correct
#
# Translation not working:
#   - Check internet connection (internal mode)
#   - Verify API endpoints are accessible
#   - Check API keys are correct
#   - Review logs for error messages
#
# Docker audio issues:
#   - Ensure host has audio devices
#   - Check device permissions
#   - Use --device /dev/snd in docker run
